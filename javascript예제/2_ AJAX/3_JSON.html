자바스크립트의 객체를 문자열로 표현하는 방법 JSON=javascript object notation

var pi = 3.14

var str = "string"

JSON.stringify(pi)

JSON.stringify(str)

JSON.parse("3.14")

JSON.parse("\"string\"")

var arr=["문자열", 3.14, true, null, undefined, function a(){console.log("method")}]

var t = JSON.stringify(arr)

JSON.parse(t)

var obj = {"str":"문자열", "num":3.14, "boolen":true, "null":null, "undefined":undefined, "method":function a(){console.log("method")}}

var t2 = JSON.stringify(obj)

JSON.parse(t2)

JSON.stringify(object)=JSON문자열로 반환하기

JSON.parse(string)=JSON문자열을 JAVASCRIPT OBJECT로 반환하기 단 undefined랑 function은 반환안됨.
